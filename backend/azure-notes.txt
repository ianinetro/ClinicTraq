Azure App Service Configuration Notes
======================================
Deployment: VM-based using ./deploy.sh (Azure CLI zip deploy)
Runtime: Python 3.11
Startup command: bash startup.sh

── One-time VM setup ────────────────────────────────────────────────────────
# Install Azure CLI
curl -sL https://aka.ms/InstallAzureCLIDeb | sudo bash
az login

# Install Node.js 20 (for frontend build + wrangler)
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo bash -
sudo apt-get install -y nodejs
npm install -g wrangler
wrangler login    # authenticates to Cloudflare

── Deploy commands ──────────────────────────────────────────────────────────
export AZURE_RESOURCE_GROUP=clinictraq-rg
export AZURE_WEBAPP_NAME=clinictraq-api
export VITE_API_URL=https://clinictraq-api.azurewebsites.net/api/v1
export CF_PROJECT_NAME=clinictraq

./deploy.sh                  # deploy both backend + frontend
./deploy.sh --backend-only
./deploy.sh --frontend-only

  DATABASE_URL        = postgresql+asyncpg://user:pass@server.postgres.database.azure.com:5432/clinictraq?ssl=require

  WEB_WORKERS         = 2
  AZURE_STORAGE_CONNECTION_STRING = <from Azure Portal → Storage Account → Access keys>
  AZURE_STORAGE_CONTAINER = clinictraq-uploads

── Azure resources to create ────────────────────────────────────────────────
  az group create --name clinictraq-rg --location eastus
  az appservice plan create --name clinictraq-plan --resource-group clinictraq-rg --sku B2 --is-linux
  az webapp create --name clinictraq-api --resource-group clinictraq-rg --plan clinictraq-plan --runtime "PYTHON:3.11"
  az postgres flexible-server create --name clinictraq-db --resource-group clinictraq-rg --sku-name Standard_B2ms --tier Burstable --version 16 --admin-user clinictraq --admin-password <password>
  az redis create --name clinictraq-cache --resource-group clinictraq-rg --sku Basic --vm-size c0
  az storage account create --name clinictraqstorage --resource-group clinictraq-rg --sku Standard_LRS --kind StorageV2

── Cloudflare Pages (frontend, built on VM + pushed via wrangler) ──────────
  Build output: frontend/dist
  deploy.sh handles: npm ci → npm run build → wrangler pages deploy dist
  _redirects handles SPA routing (already in frontend/public/)
  Add custom domain in Cloudflare Pages dashboard after first deploy.
